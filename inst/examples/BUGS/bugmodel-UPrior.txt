model{

K     ~ dunif(1.0, 100.0)
logr0    ~ dunif(-5.0, 5.0)
logtheta ~ dunif(-5.0, 5.0)
stdQ ~ dunif(0.0001,100)
stdR ~ dunif(0.0001,100)
# JAGS notation, mean, and precision ( reciprical of the variance, 1/sigma^2)
iQ <- 1/(stdQ*stdQ);
iR <- 1/(stdR*stdR);

r0 <- exp(logr0)
theta <- exp(logtheta)


x[1] ~ dunif(0,10)

for(t in 1:(N-1)){
  mu[t] <- x[t] + exp(r0 * (1 - x[t]/K)* (x[t] - theta) )
  x[t+1] ~ dnorm(mu[t],iQ) 
}


for(t in 1:(N)){
  y[t] ~ dnorm(x[t],iR)
}

}
