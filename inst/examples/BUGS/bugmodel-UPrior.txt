model{

K     ~ dunif(1.0, 22000.0)
logr0    ~ dunif(-4.0, 2.0)
logtheta ~ dunif(-4.0, 2.0)
stdQ ~ dunif(0,100)
stdR ~ dunif(0,100)
iQ <- 1/(stdQ*stdQ);
iR <- 1/(stdR*stdR);

r0 <- exp(logr0)
theta <- exp(logtheta)


x[1] ~ dunif(0,10)

for(t in 1:(N-1)){
  mu[t] <- x[t] + exp(r0 * (1 - x[t]/K)* (x[t] - theta) )
  x[t+1] ~ dnorm(mu[t],iQ)
}


for(t in 1:(N)){
  y[t] ~ dnorm(x[t],iR)
}

}
